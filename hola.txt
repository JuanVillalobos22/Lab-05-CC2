#include<iostream>
using namespace std;

int main(){
	float ph, ht, con, sueldo;
	cout<<"cuanto ganas por hora:  ";
	cin>>ph;
	cout<<"cuantas horas tabajo:  ";
	cin>>ht;
	con=(ph*ht)*7;
	if(con>=0 && con<150){
		sueldo=con-(0.05*con);
		cout<<"sueldo:  "<<sueldo;
	}
	else if(con>=150 && con<300){
		sueldo=con-(0.07*con);
		cout<<"sueldo:  "<<sueldo;
	}
	else if(con>=300 && con<450){
		sueldo=con-(0.09*con);
		cout<<"sueldo:  "<<sueldo;
	}
	else
		cout<<"el sueldo es mayor a 450, el sueldo es: "<<con;

	return 0;
}
